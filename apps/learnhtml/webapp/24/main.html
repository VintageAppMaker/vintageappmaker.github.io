개발 노트: ES5 vs ES6 예시 (같은 기능)
할 일 목록을 필터링하는 함수를 ES5/ES6로 비교합니다.
// ES5
function filterByTextES5(list, q) {
  q = (q || '').toLowerCase();
  return list.filter(function(item){
    return String(item.title).toLowerCase().indexOf(q) !== -1;
  });
}

// ES6
const filterByTextES6 = (list, q = '') =>
  list.filter(item => String(item.title).toLowerCase().includes(String(q).toLowerCase()));
