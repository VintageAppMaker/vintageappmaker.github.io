function saveAndRender() {
  // 1. 상태 저장 (LocalStorage)
  localStorage.setItem('vibe_todos', JSON.stringify(todos));
  
  // 2. 리스트 UI 업데이트
  renderList();
  
  // 3. 차트 시각화 업데이트
  renderChart();
}

function addTodo() {
  const input = document.getElementById('todoInput');
  const newTodo = {
    id: Date.now(),
    text: input.value,
    completed: false
  };
  todos.push(newTodo);
  saveAndRender();
}
